# Async alert

this.Dialog.alert({
  title: "这个是标题",
  message: "这个是内容",
  beforeClose: (e) => {// 关闭之前的回调事件
    // e代表点击了哪个事件
  },
  afterClose: () => {// 关闭之后的回调事件
    this.Toast("彻底关闭了");

});

# Async cancel, Async confirm

this.Dialog.confirm({
  title: "这个是标题",
  message: "这个是内容",
  beforeClose: (e) => {
    console.log("这个是会告诉你点击了哪个按钮", e);
    return new Promise((resolve, reject) => {
      console.log(reject);
      setTimeout(() => {
        resolve("请求结束");
      }, 3000);
    }).catch(err => {
      console.log("错误信息在这里展示", err);
    });
  }
});
