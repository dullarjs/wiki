  "beforeEach",
  (from: Route, to: Route, next: NavigationGuardNext) => {
    const token = Store.get("headerToken");

    if (
      Object.keys(token).length == 0 &&
      from.name != "login" &&
      from.name != "ecslogin"
    ) {
      const redirect = btoa(encodeURIComponent(to.path));
      next({ path: `/login?redirect=${redirect}` });
    } else {
      console.log(to);
      next({});
    }
  }
);
