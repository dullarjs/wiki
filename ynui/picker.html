# picker

  • single picker
<template>
  <div>
    <yn-button type="primary" @click="handleClick(1)">single picker</yn-button>
    <yn-picker
      v-model="show1"
      :columns="columns1"
      @confirm="handleConfirm"
      @change="handleChange"
    ></yn-picker>
  </div>
</template>
<script type="text/javascript">
export default {
  name: "DemoPicker",
  props: {
    demo: String
  },
  data() {
    return {
      show1: false,
      columns1: [
        "Shanghai",
        "Toyko",
        "Beijing",
        "NewYork",
        "Shenzhen",
        "Hongkong"
      ]
    };
  },
  methods: {
    handleChange(pick, index) {
      this.Toast(`选择第${index}个，值为${pick.value || pick}`);
    },
    handleConfirm(e) {
      console.log(e);
      this.Toast(JSON.stringify(e));
    },
    handleClick(i) {
      this[`show${i}`] = !this[`show${i}`];
    }
  }
};
</script>
